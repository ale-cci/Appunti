\begin{minipage}{0.5\textwidth}
    \begin{itemize}
        \item OFF
            \[ I_{DS} = 0 \qquad \text{per}\, V_{GS} < V_{Tn} \]
        \item SATURAZIONE
            \[ I_D  = \frac{\beta}{2} {(V_{GS} - V_{Tn})}^2 \quad\text{per}
            \begin{cases}
                V_{DS} \ge V_{GS} - V_{Tn}\\
                V_{GS} \ge V_{Tn}
            \end{cases}
        \]
        \item LINEARITÀ
            \[
                I_D = \beta\left((V_{GS} - V_T)V_{DS} - \frac{V_{DS}^2}{2}\right)
                \quad\text{per}
                \begin{cases}
                    V_{DS} < V_{GS} - V_{Tn}\\
                    V_{GS} \ge V_{Tn}
                \end{cases}
            \]
    \end{itemize}
\end{minipage}
